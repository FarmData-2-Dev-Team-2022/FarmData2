/**
 * A Vue component for a color-changing input form.
 *
 * 
 * @vue-prop {String} defaultInput - The default display value of the input form
 * @vue-prop {String} firstC - The default color of the first circle element.
 * @vue-prop {String} secondC - The default color of the second circle element.
 * @vue-prop {String} thirdC - The default color of the third circle element. 
 * 
 * @vue-event {String} input changed - Displays changed input as a text on the bottom and check if the input is a CSS color; if it is, the color of the input form.
 * @vue-event {String} circle input changed - changes the color of the circles and emits the color value to the parent.
 * @vue-event resetInput - clears the input field when clicked.
 */ 
let CoolInputFormComponent = {
    template: `<span>
            <slot></slot>
            <input type="text" id="cool" :style="{'background-color': selectedColor}" @click="resetInput" v-model="typedInput">
            <p>{{typedInput}}</p>   
            <br>
            <input
                type="text"
                v-model="firstColor"
                @click="resetInput"
            />
            <input
                type="text"
                v-model="secondColor"
                @click="resetInput"
            />
            <input
                type="text"
                v-model="thirdColor"
                @click="resetInput"
            />
            <br>
            <div v-bind:style="circle1"></div>
            <div v-bind:style="circle2"></div>
            <div v-bind:style="circle3"></div>
            </span>`,
    props: {
            defaultInput:{
                type: String,
            }, 
            firstC: {
                type: String,
            }, 
            secondC: {
                type: String,
            }, 
            thirdC: {
                type: String,
            }


    },
    data(){
        return {

            selectedColor: '',
            typedInput: this.defaultInput,
            circle1: {
                backgroundColor: this.firstC,
                borderRadius: '50%',
                width: '100px',
                height: '100px',
                margin: '40px',
                float: 'left',
            },
            circle2: {
                backgroundColor: this.secondC,
                borderRadius: '50%',
                width: '100px',
                height: '100px',
                margin: '40px',
                float: 'left',
            },
            circle3: {
                backgroundColor: this.thirdC,
                borderRadius: '50%',
                width: '100px',
                height: '100px',
                margin: '40px',
                float: 'left',
            },
            firstColor: '', 
            secondColor: '',
            thirdColor: '',
        } 
    },
    methods: {
        resetInput(){
            this.typedInput = ''
        },
        addColorArray(color){
            this.$emit('update-color', color)
        }
    },
    watch: {
        typedInput(newTypedValue){
            var s = new Option().style;
            s.color = newTypedValue;
            if ( s.color == newTypedValue.toLowerCase()) {
                this.selectedColor = newTypedValue;
            } else {
                this.selectedColor = '';
            }
        },
        firstColor(newColor) {
            var s = new Option().style;
            s.color = newColor;
            if ( s.color == newColor.toLowerCase() && newColor !="") {
                this.firstColor = newColor;
                this.circle1.backgroundColor = newColor;
                this.addColorArray(newColor)
            }
        },
        secondColor(newColor) {
            var s = new Option().style;
            s.color = newColor;
            if ( s.color == newColor.toLowerCase() && newColor !="") {
                this.secondColor = newColor;
                this.circle2.backgroundColor = newColor;
                this.addColorArray(newColor)
            }
        },
        thirdColor(newColor) {
            var s = new Option().style;
            s.color = newColor;
            if ( s.color == newColor.toLowerCase() && newColor !="") {
                this.thirdColor = newColor;
                this.circle3.backgroundColor = newColor;
                this.addColorArray(newColor);
            }
        },
    }

}

    
try {
    module.exports = {
        CoolInputFormComponent: CoolInputFormComponent
    }
}
catch {}
